Config {
    Monitor 9600;
    Saida 11;
    Entrada A0;
}
real sensorAtivo;

Repita {
    sensorAtivo = Leia A0;
    
    se (sensorAtivo >= 0.4) {
        EscritaMon("Sensor ativado!");
        Escreva 11 1;
    } senao {
        EscritaMon("Sensor desativado");
        Escreva 11 0;
    }

    EscritaMon(sensorAtivo);
    Espere 1000 ms;
}
